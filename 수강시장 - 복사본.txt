        function bigsubjectchange(item) { //두번째 select box
            var temp, i = 0, j = 0;
            var ccount, cselect;

            temp = document.signform.big_subject;

            for (i = (temp.options.length - 1); i > 0; i--) {
                temp.options[i] = null;
            }

            eval('ccount = big_subject' + item + '.length');

            for (j = 0; j < ccount; j++) {
                eval('cselect = big_subject' + item + '[' + j + '];');
                temp.options[j] = new Option(cselect, cselect);
            }
            temp.options[0].selected = true;
            return true;
        }
